\begin{definition}
  \lean{polar}
  Let $f$ be a function $f : M \to N$. The polar of $f$ is $f(x + y) - f(x) - f(g)$.
\end{definition}

\begin{theorem}
  \lean{map_add}
  We have the following equality.
$$
  f(x + y) = f(x) + f(y) + \mathrm{Polar} f(x,y)
$$
\end{theorem}
